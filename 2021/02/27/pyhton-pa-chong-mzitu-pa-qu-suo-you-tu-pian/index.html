2.网站分析

观察页面

  • 每个页面都有一些封面图片，而且页面是规律的递增。

  • 也就是爬取每个页面封面<a>标签的属性，然后进入每个a的属性链接，爬取页面的图片

获取页面单个图片封面的a标签href属性
import requests
from lxml import etree

headers = {
    'User-Agent':'Mozilla/5.0 (Windows NT 6.1; WOW64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/56.0.2924.87 Safari/537.36'

res = requests.get("https://www.mzitu.com/",headers = headers)

二、代码逻辑编写

静态网站，很好写爬虫。

  • 获取详细页面的图片是，res2掉了个2，导致获取图片都是一样的，以为有时什么反爬机制。
  • 站内的地址，在浏览器中是不能访问的，需要下载下来。
  • xpath选择的属性，即使只有一个值返回的也是列表
  • 代码写好了，下载太慢了，要使用多线程。现学一波

1.多线程学习

  • 学习地址，官方文档：https://docs.python.org/zh-cn/3.7/library/threading.html

  • 还看了一下其他的博客，写得不怎么好，有的有点帮助，博客地址：https://blog.csdn.net/qq_40244755/article/details/90043484

  • target就是指定的函数名字，args就是给函数传入的参数

    示例代码

    两个函数同时执行的，一共花了六秒程序执行完成，如果不用多线程要花12秒。这个完成是在第二行输出的，可见函数从入口进去了，后面的代码就继续执行。在函数的最后输出完成信息带提示这个线程执行完成了。知道这些就可以了，接下来硬核使用多线程。

from threading import Thread
import time

def test_1(start,end):
    for i in range(start,end):
        print("当前爬取第%s页"%(i))
        time.sleep(2)
        
t1 = Thread(target=test_1,args=(0,3))
t2 = Thread(target=test_1,args=(3,6))
t1.start()
t2.start()
print("完成")

2.最终代码

写好了执行出了错误，最后以看，原来网站不能访问了。

import requests
from lxml import etree
import os


MAX_PAGES = 3   #一共260页
main_link = "https://www.mzitu.com/page/"

headers = {
            "User-Agent": "Mozilla/5.0 (Windows NT 10.0; WOW64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/70.0.3538.110 Safari/537.36",
            "Referer": "https://www.mzitu.com/xinggan/"
}
#从总页面获取每个详细页面
def get_page(currentPage):
    res = requests.get(currentPage,headers = headers)
    demo = etree.HTML(res.text)
    detail_url = demo.xpath('//*[@id="pins"]/li/a/@href')
    detail_name = demo.xpath('//*[@id="pins"]/li/span/a/text()')
    # 获取到的名字和地址都是列表类型，打包为字典类型
    detail  = dict(zip(detail_name,detail_url))
    for key,value in detail.items():
        get_detail(key,value)

#从详细页面获取每一张图片的地址
def get_detail(name,url):

    res = requests.get(url,headers = headers)
    demo = etree.HTML(res.text)
    allCounts = demo.xpath('/html/body/div[2]/div[1]/div[4]/a[5]/span/text()')
    allCounts = int(allCounts[0])
    
    
    for i in range(1,allCounts+1):
        path = url+'/'+str(i)
        res2 = requests.get(path,headers = headers)
        demo2 = etree.HTML(res2.text)
        oneIMG = demo2.xpath('//img[@class="blur"]/@src')
        download_img(name,oneIMG[0],i)
    
def download_img(name,url,count):
    if not os.path.exists("/home/mzitu/%s"%(name)):
        os.mkdir("/home/mzitu/%s"%(name))
    res = requests.get(url,headers = headers)
    f = open('/home/mzitu/%s/%s.jpg'%(name,count),'wb+')
    f.write(res.content)
    f.close()
    
    
def start_spider(start,end):
    url = "https://www.mzitu.com/page/"
    for i in range(start,end):
        print("------------线程%s--%s开始爬取页面%s-------------"%(start,end,i))
        currentURL = url+str(i)
        get_page(currentURL)
    print("------------线程%s--%s结束-------------"%(start,end))
    

if __name__=='__main__':
    start_spider(1,21)

4.最终代码（添加代理）

import requests
from lxml import etree
import os
import csv

MAX_PAGES = 3   #一共260页
main_link = "https://www.mzitu.com/page/"

headers = {
            "User-Agent": "Mozilla/5.0 (Windows NT 10.0; WOW64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/70.0.3538.110 Safari/537.36",
            "Referer": "https://www.mzitu.com"
}

proxies = {
    "http":"http://127.0.0.1:1081",
    "https":"http://127.0.0.1:1081"
}
#从总页面获取每个详细页面
def get_page(currentPage):
    try:
        res = requests.get(currentPage,headers = headers,proxies=proxies)
        demo = etree.HTML(res.text)
        detail_url = demo.xpath('//*[@id="pins"]/li/a/@href')
        detail_name = demo.xpath('//*[@id="pins"]/li/span/a/text()')
    # 获取到的名字和地址都是列表类型，打包为字典类型
        detail  = dict(zip(detail_name,detail_url))
        for key,value in detail.items():
            get_detail(key,value)
    except:
        print("网络问题，详细集合界面出错:%s"%(currentPage))

#从详细页面获取每一张图片的地址
def get_detail(name,url):
    res = requests.get(url,headers = headers,proxies=proxies)
    demo = etree.HTML(res.text)
    allCounts = demo.xpath('/html/body/div[2]/div[1]/div[4]/a[5]/span/text()')
    allCounts = int(allCounts[0])
    
    
    for i in range(1,allCounts+1):
        path = url+'/'+str(i)
        try:
            res2 = requests.get(path,headers = headers,proxies=proxies)
            demo2 = etree.HTML(res2.text)
            oneIMG = demo2.xpath('//img[@class="blur"]/@src')
            download_img(name,oneIMG[0],i)
        except:
            print(path)
            print(oneIMG)

    
def download_img(name,url,count):
    if not os.path.exists("./mzitu/%s"%(name)):
        os.mkdir("./mzitu/%s"%(name))
    res = requests.get(url,headers = headers,proxies=proxies)
    f = open('./mzitu/%s/%s.jpg'%(name,count),'wb+')
    f.write(res.content)
    f.close()
    
    
def start_spider(start,end):
    url = "https://www.mzitu.com/page/"
    for i in range(start,end):
        print("------------线程%s--%s开始爬取页面%s-------------"%(start,end,i))
        currentURL = url+str(i)
        get_page(currentURL)
    print("------------线程%s--%s结束-------------"%(start,end))
    

if __name__=='__main__':
    start_spider(1,2)

三、总结

  • 其实这个博客看起来不是很详细，详细的是代码，可以从代码中学习。感觉这次代码写得还是比较清晰明了，算是函数式编程吧，但也有些不足，因为逻辑关系速度不是很快，可以优化的。
  • 但最后发现是网速原因，也就没去优化什么的了，代理下载实在是太慢了，也没什么必要全部下载下来（要是有个快点的代理就好了）
  • 这次项目新学到的是使用代理，其他的得到了联系，下次写代码的时候先要考虑一下性能，至于自己下载不下载得完所有，待补充吧
