# coding:utf-8
from django.shortcuts import render
from django.http import HttpResponse
 
# 引入我们创建的表单类
from .forms import AddForm
 
def index(request):
    if request.method == 'POST':# 当提交表单时
     
        form = AddForm(request.POST) # form 包含提交的数据
         
        if form.is_valid():# 如果提交的数据合法
            a = form.cleaned_data['a']
            b = form.cleaned_data['b']
            return HttpResponse(str(int(a) + int(b)))
     
    else:# 当正常访问时
        form = AddForm()
    return render(request, 'index.html', {'form': form})

class PostForm(forms.ModelForm):

Session

参考地址：https://code.ziqiangxuetang.com/django/django-session.html

  • session 默认有效时间为两周，可以在 settings.py 中修改默认值
# 创建或修改 session：
request.session[key] = value
# 获取 session：
request.session.get(key,default=None)
# 删除 session
del request.session[key] # 不存在时报错
  • 一个简化的登陆认证
def login(request):
    m = Member.objects.get(username=request.POST['username'])
    if m.password == request.POST['password']:
        request.session['member_id'] = m.id
        return HttpResponse("You're logged in.")
    else:
        return HttpResponse("Your username and password didn't match.")
         
         
def logout(request):
    try:
        del request.session['member_id']
    except KeyError:
        pass
    return HttpResponse("You're logged out.")

当登陆时验证用户名和密码，并保存用户id在 session 中，这样就可以在视图中用 request.session[‘member_id’]来检查用户是否登陆，当退出的时候，删除掉它。

cookie

http协议是无状态的，cookie存储在浏览器中，浏览器从某个服务器请求页面时，会向服务器会送之前收到的cookie

  • 一般开发使用高层框架session或者user框架。
  • cookies信息存储在本地，用户可以拒绝，也容易被嗅探到。通过http协议发送（明文）

django提供了相关工具来处理用户注册和认证功能。

使用步骤
  • 确定安装session框架，用户使用cookie
  • ‘django.contrib.auth’放在INSTALLED_APPS中，运行 manage.py syncdb创建对应的数据表
  • MIDDLEWARE_CLASSES中包含django.contirb.auth.middkeware.AuthenticationMiddleware
使用

登录和退出

Django提供两个函数来执行django.contrib.auth中的动作: authenticate()和login()

  • 认证给出用户名和密码。authenticate()函数接受两个参数，不符合返回None
from django.contrib import auth
user = auth.authentiate(username = "join",passwort="secret")
if user is not Noen:
		print("Correct")
else:
		print("Invalid password")
  • login()函数接受一个HttpRequest对象和一个User对象作为参数并使用Django的会话框架把用户的ID保存在该会话中
一个视图中使用authenticate()和login()
from django.contrib import auth

def login_views(request):
		username = request.POST.get('username','')
		password = request.POST.get('password','')
		user = auth.authenticate(username=username,password=password)

注销用户

注销哟呵用户，在视图中只用djano.contrib.logout()，它接受一个HttpRequset对象并且没有返回值

from django.contrib import auth

def logout_view(request):
		auth.logout(request)
		return HttpResponseRedirect("/account/loginedout")

即使用户没有登录也不会报错
